import React, { useState, useEffect } from 'react';
import { 
  Users, BookOpen, Calendar, TrendingUp, 
  DollarSign, Bell, Settings, LogOut, Menu,
  GraduationCap, Bus, Heart, Camera
} from 'lucide-react';
import './modern-theme.css';

const ModernDashboard = () => {
  const [sidebarOpen, setSidebarOpen] = useState(true);
  const [stats, setStats] = useState({
    students: 0,
    teachers: 0,
    attendance: 0,
    revenue: 0
  });
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchDashboardStats();
  }, []);

    try {
      const token = localStorage.getItem('token');
      const response = await fetch('/api/dashboard/stats', {
        headers: { 'Authorization': `Bearer ${token}` }
      });
      const data = await response.json();
      if (data.success) {
        setStats(data.stats);
      }
    } catch (error) {
      console.error('Failed to fetch stats:', error);
    } finally {
      setLoading(false);
    }
  };

    { icon: TrendingUp, label: 'Dashboard', active: true },

    { icon: Camera, label: 'CCTV', active: false },
    { icon: Heart, label: 'Health', active: false },
    { icon: Settings, label: 'Settings', active: false },
  ];

  const statCards = [
    { 
      icon: Users, 
      value: stats.students, 
      label: 'Total Students', 
      change: '+12%',
      changeType: 'positive',

    { 
      icon: GraduationCap, 
      value: stats.teachers, 
      label: 'Teachers', 
      change: '+5%',
      changeType: 'positive',
      color: 'pink'

    { 
      icon: Calendar, 
      value: `${stats.attendance}%`,

      change: '-2%',
      changeType: 'negative',
      color: 'green'
    },
    { 
      icon: DollarSign, 
      value: `â‚¹${stats.revenue}`, 
      label: 'Revenue', 
      change: '+18%',
      changeType: 'positive',
      color: 'orange'
    },
  ];

  return (
    <div className="min-h-screen flex">
      <div className="app-background"></div>

      {/* Sidebar */}
      <aside className={`sidebar-modern transition-transform duration-300 ${sidebarOpen ? '' : '-translate-x-full'}`}>
        <div className="sidebar-logo">
          <div className="sidebar-logo-icon">ðŸ“š</div>

        </div>

          {navItems.map((item, index) => (

              key={index}

              className={`sidebar-nav-item ${item.active ? 'active' : ''}`}
            >
              <item.icon className="sidebar-nav-icon" />
              <span>{item.label}</span>
            </a>
          ))}

          <button className="sidebar-nav-item w-full text-red-400 hover:text-red-300">
            <LogOut className="sidebar-nav-icon" />
            <span>Logout</span>
          </button>
        </div>

            <Menu className="w-6 h-6" />
